#include <iostream>
#include <iomanip>

int main() {
    float a1 = 1.23456789f; // 4 Bytes. float numbers end with 'f' otherwise they are treated as double by default.
    float a2 = 123456789.123456789f;
    float a3 = 1234.56789f;
    std::cout << std::setprecision(10) << a1 << " "<< a2 << " "<< a3 << std::endl; // precision upto 7 digits only possible for float.

    double b1 = 1.2345678901234567890; // 8 Bytes. double numbers are default for decimal numbers.
    double b2 = 12345678901234567890.12345678901234567890;
    double b3 = 1234567890.1234567890;
    std::cout << std::setprecision(20) << b1 << " " << b2 << " " << b3 << std::endl; // precision upto 15 digits only possible for double.
    
    long double c1 = 1.23456789012345678901234567890L; // 12 Bytes. long double numbers end with 'L' otherwise they are treated as double by default.
    long double c2 = 12345678901234567890.12345678901234567890L;
    long double c3 = 1234567890.12345678901234567890L;
    std::cout << std::setprecision(25) << c1 << " " << c2 << " " << c3 << std::endl; // more precise than double.

    double d1{5.6};
    double d2{-5.6};
    double d3{};
    double d4{};

    std::cout << d1/d3 << " " << d2/d3 << " " << d3/d4 << " " << d1/d3 + d1 << " " << d2/d3 + d1 << " " << d3/d4 + d1 << std::endl; // output will be : inf -inf -nan inf -inf -nan

    return 0;
}
